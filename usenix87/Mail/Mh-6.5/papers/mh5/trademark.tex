% Trademarks...

\def\gobble#1{}

\def\trademark#1#2{%
    \let#1/\relax			% in case #2 references #1
    \setbox0=\hbox{#2}%			%   when we make the \hbox
    \ifdim\wd0>0pt
	\gdef#1/{%
	    \trademark{#1}\empty	% only in TeX: a recursive macro
	    #1/\nfootnote{#1/ is a trademark of #2.}%
	}%
    \else
	\gdef#1/{{\sc\expandafter\gobble\string#1}}%
    \fi
}
